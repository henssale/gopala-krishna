        // --- Global State ---
        let cartItems = [];

        // --- DOM Elements ---
        const productListDiv = document.getElementById('product-list');
        const cartButton = document.getElementById('cart-button');
        const cartItemCountSpan = document.getElementById('cart-item-count');
        const cartModal = document.getElementById('cart-modal');
        const closeCartButton = document.getElementById('close-cart-button');
        const cartItemsContainer = document.getElementById('cart-items-container');
        const emptyCartMessage = document.getElementById('empty-cart-message');
        const cartTotalItemsSpan = document.getElementById('cart-total-items');
        const cartTotalPriceSpan = document.getElementById('cart-total-price');
        const checkoutButton = document.getElementById('checkout-button');
        const currentYearSpan = document.getElementById('current-year');

        // --- Utility Functions ---

        // Formats a number as currency (USD)
        const formatCurrency = (amount) => {
            return new Intl.NumberFormat('en-IN', { // Changed 'rupees' to 'INR' for Indian Rupees
                style: 'currency',
                currency: 'INR',
            }).format(amount);
        };

        // Updates the cart item count badge in the header
        const updateCartCountBadge = () => {
            const totalCount = cartItems.reduce((sum, item) => sum + item.quantity, 0);
            cartItemCountSpan.textContent = totalCount;
            if (totalCount > 0) {
                cartItemCountSpan.classList.remove('hidden');
            } else {
                cartItemCountSpan.classList.add('hidden');
            }
        };

        // Renders all product cards on the page
        const renderProductCards = () => {
            productListDiv.innerHTML = ''; // Clear existing cards
            initialProducts.forEach(product => {
                const productCardHtml = `
                    <div class="product-card">
                        <img
                            src="${product.imageUrl}"
                            alt="${product.name}"
                            class="w-full h-48 object-cover rounded-t-xl"
                            onerror="this.onerror=null; this.src='https://placehold.co/400x300/F5F5DC/333?text=${encodeURIComponent(product.name)}';"
                        />
                        <div class="p-6 flex flex-col flex-grow">
                            <h3 class="text-xl font-semibold text-gray-800 mb-2">${product.name}</h3>
                            <p class="text-gray-600 text-sm mb-2 flex-grow">${product.description}</p>
                            <p class="text-gray-700 text-sm mb-1"><span class="font-medium">Breed:</span> ${product.breed}</p>
                            <p class="text-gray-700 text-sm mb-4"><span class="font-medium">Age:</span> ${product.age}</p>
                            <div class="flex items-baseline mb-3">
                                <span class="text-green-600 font-bold text-2xl">${formatCurrency(product.price)}</span>
                            </div>
                            <p class="text-gray-700 text-sm mb-4">Min. Order: ${product.minOrderQuantity} unit${product.minOrderQuantity > 1 ? 's' : ''}</p>
                            <div class="flex items-center mb-4">
                                <label for="quantity-${product.id}" class="mr-2 text-gray-700">Quantity:</label>
                                <input
                                    type="number"
                                    id="quantity-${product.id}"
                                    min="${product.minOrderQuantity}"
                                    value="${product.minOrderQuantity}"
                                    class="w-24 p-2 border border-gray-300 rounded-md focus:ring-green-500 focus:border-green-500"
                                />
                            </div>
                            <button
                                data-product-id="${product.id}"
                                class="add-to-cart-btn mt-auto btn-primary"
                            >
                                Add to Cart
                            </button>
                        </div>
                    </div>
                `;
                productListDiv.insertAdjacentHTML('beforeend', productCardHtml);
            });

            // Attach event listeners to all "Add to Cart" buttons
            document.querySelectorAll('.add-to-cart-btn').forEach(button => {
                button.addEventListener('click', (event) => {
                    const productId = event.target.dataset.productId;
                    const product = initialProducts.find(p => p.id === productId);
                    const quantityInput = document.getElementById(`quantity-${productId}`);
                    const quantity = parseInt(quantityInput.value);
                    addToCart(product, quantity);
                });
            });
        };

        // Adds a product to the cart or updates its quantity
        const addToCart = (product, quantity) => {
            const existingItemIndex = cartItems.findIndex(item => item.id === product.id);

            if (existingItemIndex > -1) {
                cartItems[existingItemIndex].quantity += quantity;
            } else {
                cartItems.push({ ...product, quantity: quantity });
            }
            updateCartCountBadge();
            renderCartItems(); // Re-render cart when items change
        };

        // Removes an item from the cart
        const removeFromCart = (productId) => {
            cartItems = cartItems.filter(item => item.id !== productId);
            updateCartCountBadge();
            renderCartItems(); // Re-render cart when items change
        };

        // Updates the quantity of an item in the cart
        const updateCartQuantity = (productId, newQuantity) => {
            cartItems = cartItems.map(item => {
                if (item.id === productId) {
                    // Ensure quantity doesn't go below minOrderQuantity
                    const quantity = Math.max(newQuantity, item.minOrderQuantity);
                    return { ...item, quantity: quantity };
                }
                return item;
            }).filter(item => item.quantity > 0); // Remove if quantity becomes 0 (though buttons prevent this for minOrderQuantity 1)
            updateCartCountBadge();
            renderCartItems(); // Re-render cart when items change
        };

        // Renders the items inside the cart modal
        const renderCartItems = () => {
            cartItemsContainer.innerHTML = ''; // Clear existing items
            let totalItemsInCart = 0;
            let totalPriceInCart = 0;

            if (cartItems.length === 0) {
                emptyCartMessage.classList.remove('hidden');
                checkoutButton.disabled = true;
            } else {
                emptyCartMessage.classList.add('hidden');
                checkoutButton.disabled = false;
                cartItems.forEach(item => {
                    totalItemsInCart += item.quantity;
                    totalPriceInCart += item.price * item.quantity;

                    const cartItemHtml = `
                        <li class="flex items-center bg-gray-50 p-4 rounded-lg shadow-sm">
                            <img
                                src="${item.imageUrl}"
                                alt="${item.name}"
                                class="cart-item-img mr-4"
                            />
                            <div class="flex-grow">
                                <h3 class="font-semibold text-gray-800">${item.name}</h3>
                                <p class="text-gray-600 text-sm">
                                    ${formatCurrency(item.price)} x ${item.quantity} = ${formatCurrency(item.price * item.quantity)}
                                </p>
                                <p class="text-gray-500 text-xs">Min. Order: ${item.minOrderQuantity} unit${item.minOrderQuantity > 1 ? 's' : ''}</p>
                            </div>
                            <div class="flex items-center space-x-2">
                                <button
                                    data-product-id="${item.id}"
                                    data-action="decrement"
                                    class="cart-quantity-btn btn-secondary"
                                    ${item.quantity <= item.minOrderQuantity ? 'disabled' : ''}
                                >
                                    -
                                </button>
                                <span class="font-medium text-gray-800 w-8 text-center">${item.quantity}</span>
                                <button
                                    data-product-id="${item.id}"
                                    data-action="increment"
                                    class="cart-quantity-btn btn-secondary"
                                >
                                    +
                                </button>
                                <button
                                    data-product-id="${item.id}"
                                    class="remove-from-cart-btn ml-4 btn-danger"
                                    aria-label="Remove item from cart"
                                >
                                    <svg xmlns="http://www.w3.org/2000/svg" class="h-5 w-5" viewBox="0 0 20 20" fill="currentColor">
                                        <path fill-rule="evenodd" d="M9 2a1 1 0 00-.894.553L7.382 4H4a1 1 0 000 2v10a2 2 0 002 2h8a2 2 0 002-2V6a1 1 0 100-2h-3.382l-.724-1.447A1 1 0 0011 2H9zM7 8a1 1 0 012 0v6a1 1 0 11-2 0V8zm6 0a1 1 0 11-2 0v6a1 1 0 112 0V8z" clip-rule="evenodd" />
                                    </svg>
                                </button>
                            </div>
                        </li>
                    `;
                    cartItemsContainer.insertAdjacentHTML('beforeend', cartItemHtml);
                });

                // Attach event listeners for quantity buttons and remove buttons in the cart
                document.querySelectorAll('.cart-quantity-btn').forEach(button => {
                    button.addEventListener('click', (event) => {
                        const productId = event.target.dataset.productId;
                        const action = event.target.dataset.action;
                        const item = cartItems.find(i => i.id === productId);
                        if (item) {
                            if (action === 'increment') {
                                updateCartQuantity(productId, item.quantity + 1);
                            } else if (action === 'decrement') {
                                updateCartQuantity(productId, item.quantity - 1);
                            }
                        }
                    });
                });

                document.querySelectorAll('.remove-from-cart-btn').forEach(button => {
                    button.addEventListener('click', (event) => {
                        const productId = event.target.dataset.productId;
                        removeFromCart(productId);
                    });
                });
            }

            cartTotalItemsSpan.textContent = totalItemsInCart;
            cartTotalPriceSpan.textContent = formatCurrency(totalPriceInCart);
        };

        // --- Event Listeners ---

        // Open cart modal
        cartButton.addEventListener('click', () => {
            cartModal.classList.remove('hidden');
            renderCartItems(); // Ensure cart items are up-to-date when modal opens
        });

        // Close cart modal
        closeCartButton.addEventListener('click', () => {
            cartModal.classList.add('hidden');
        });

        // Close modal if clicking outside content
        cartModal.addEventListener('click', (event) => {
            if (event.target === cartModal) {
                cartModal.classList.add('hidden');
            }
        });

        // Checkout button placeholder
        checkoutButton.addEventListener('click', () => {
            // In a real application, this would lead to a checkout page or payment gateway
            alert('Proceeding to checkout! (This is a placeholder action)');
            cartItems = []; // Clear cart after "checkout"
            updateCartCountBadge();
            renderCartItems();
            cartModal.classList.add('hidden');
        });

        // --- Initial Load ---
        document.addEventListener('DOMContentLoaded', () => {
            renderProductCards();
            updateCartCountBadge(); // Initialize badge on load
            currentYearSpan.textContent = new Date().getFullYear(); // Set current year in footer
            console.log("Happy Hens Farm website loaded successfully!");
        });
